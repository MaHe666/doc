        # 第一个示例

        注意：名为 main 的package 代表该程序可独立执行，每个 Go 应用程序都包含一个名为 main 的包

        注意：{ 符号不能单独放在一行

        注意：+ 只能用作字符串之间拼接

        package main

import (
	"fmt"
	"time"
)

func main() {
	var s string = "Hello"
	fmt.Println(s + "World")
	fmt.Println("当前时间：", time.Now())
}
