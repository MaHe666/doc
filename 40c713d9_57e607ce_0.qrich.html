package com.mahe666;


import org.apache.poi.hssf.usermodel.HSSFWorkbook;

import org.apache.poi.ss.usermodel.Cell;

import org.apache.poi.ss.usermodel.Row;

import org.apache.poi.ss.usermodel.Sheet;

import org.apache.poi.ss.usermodel.Workbook;

import org.apache.poi.xssf.usermodel.XSSFWorkbook;

import org.joda.time.DateTime;

import org.junit.Test;


import java.io.FileOutputStream;

import java.io.IOException;


/**

* @author ：mahe666

* @description：ExcelWriteTest

* @date ：Created in 2021/9/4 9:29

*/

public class ExcelWriteTest01 {


final static String PATH = "C:\\Users\\30704\\Desktop\\";


// final String PATH = "D:\\Code\\IDEA\\excelTest\\poi-test\\";



@Test

//测试写入03版本的excel

public void testWrite03() throws IOException {

//创建一个工作簿，使用Excel能操作的他都可以操作

Workbook workbook = new HSSFWorkbook(); //03

//创建一个工作表

Sheet sheet = workbook.createSheet("testWrite03表"); //通过工作簿创建工作表，参数为表名

//创建一个行

Row row1 = sheet.createRow(0); //通过工作表创建行，0代表第一行

//创建一个单元格

Cell cell11 = row1.createCell(0); //选中单元格，第一列 (1,1)

cell11.setCellValue("giao1"); //设置值


Cell cell12 = row1.createCell(1); //选中单元格，第二列 (1,2)

cell12.setCellValue("giaogiao2");


Row row2 = sheet.createRow(1); //1代表第二行

Cell cell21 = row2.createCell(0); //(2,1)

cell21.setCellValue("时间");

Cell cell22 = row2.createCell(1); //(2,2)

//joda-time这个工具类比官方的java.util.Date好用

cell22.setCellValue(new DateTime().toString("yyyy-MM-dd HH:mm:ss"));


//生成一张工作簿（IO流） 03版本使用xls结尾

FileOutputStream fileOutputStream = new FileOutputStream(PATH+"poi-test03.xls");

//输出

workbook.write(fileOutputStream);

//关闭流

fileOutputStream.close();

System.out.println("poi-test03.xls 文件生成完毕");

}
