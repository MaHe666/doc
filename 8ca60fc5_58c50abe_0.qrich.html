class Demo {

constructor(a, b) {

this.a = a;

this.b = b;

console.log('Demo');

}

sum() {

return this.a + this.b;

}

}

let demo1 = new Demo(2, 1);

let demo2 = new Demo(3, 1);

// 两者原型链是相等的

console.log(demo1._proto_ == demo2._proto_); // true

demo1._proto_.sub = function() {

return this.a - this.b;

}

console.log(demo1.sub()); // 1

console.log(demo2.sub()); // 2
