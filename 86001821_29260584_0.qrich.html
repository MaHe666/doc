<!DOCTYPE mapper

PUBLIC "-//mybatis.org//DTD Mapper 3.0//EN"

"http://mybatis.org/dtd/mybatis-3-mapper.dtd">

<!--namespace 绑定对应的mapper接口-->

<mapper namespace="com.mahe666.mapper.UserMapper">

<!--select查询语句块-->

<select id="getUserList" resultType="com.mahe666.pojo.User">

select * from `user`;

</select>

<!--#号内的参数名为你传递过来的参数名-->

<select id="getUserById" parameterType="int" resultType="com.mahe666.pojo.User">

select * from `user` where id = #{id}

</select>


<!--对象中的属性可以直接取出-->

<insert id="addUser" parameterType="com.mahe666.pojo.User">

insert into `user` (id,name,pwd) values (#{id},#{name},#{pwd})

</insert>


<update id="updateUser" parameterType="com.mahe666.pojo.User">

update `user` set name=#{name} , pwd=#{pwd} where id=#{id};

</update>


<!--参数类型为int的可以不写-->

<delete id="deleteUser" parameterType="int">

delete from `user` where id=#{id}

</delete>


</mapper>
