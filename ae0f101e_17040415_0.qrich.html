//封装axios

import axios from 'axios'

import qs from 'qs'

const instance = axios.create({

// 设置请求地址

baseURL:'',

timeout:6000, //超时

})



//请求拦截器

instance.interceptors.request.use(function (config) {

// 发送请求前将参数利用qs转化

config.data = qs.stringify(config.data)

return config;

});



export default instance
