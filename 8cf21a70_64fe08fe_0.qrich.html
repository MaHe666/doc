package com.mahe666.springcloud.controller;


import com.mahe666.springcloud.pojo.Dept;

import org.springframework.beans.factory.annotation.Autowired;

import org.springframework.web.bind.annotation.*;

import org.springframework.web.client.RestTemplate;


import java.util.List;


@RestController

@RequestMapping("/consumer/dept")

public class DeptConsumerController {

//消费者，不应该有Service层


//使用Ribbon之后，我们这里的地址应该是一个变量，通过服务名来访问

//private static final String REST_URL_PREFIX = "http://localhost:8001/dept";

//这里的名字是配置文件中Spring的服务名，都调用一个名字，已达到负载均衡的效果

private static final String REST_URL_PREFIX = "http://SPRINGCLOUD-PROVIDER-DEPT/dept";


@Autowired

private RestTemplate restTemplate;


@PostMapping("/add")

public Boolean add(Dept dept){

return restTemplate.postForObject(REST_URL_PREFIX+"/add",dept,Boolean.class);

}


@GetMapping("/selectById/{id}")

public Dept selectById(@PathVariable("id") Long id){

return restTemplate.getForObject(REST_URL_PREFIX+"/selectById/"+id ,Dept.class);

}


@GetMapping("/selectAll")

public List<Dept> selectAll(){

return restTemplate.getForObject(REST_URL_PREFIX+"/selectAll/" ,List.class);

}


@GetMapping("/discovery")

public Object discovery(){

return restTemplate.getForObject(REST_URL_PREFIX+"/discovery",Object.class);

}


}
