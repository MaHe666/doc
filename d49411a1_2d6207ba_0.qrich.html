// 1.使用 forEach

let myMap = new Map();

myMap.set(0, "zero");

myMap.set(1, "one");

// 0 = zero , 1 = one

myMap.forEach(function(value, key) {

console.log(key + " = " + value);

}, myMap)


// 2. 也可以使用 for...of






配合箭头函数


this.map.forEach((value, key) => {

this.list.push({

name: key,

pwd: value

});

})
