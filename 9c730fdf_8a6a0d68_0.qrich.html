package com.mahe666.pc;


/**

* 线程之间的通信问题：生产者和消费者问题

* <p>

* 线程交替执行 让A和B操作同一个变量

* int num = 0;

* A执行完num+=1后，B再执行num-=1

* 如果两个线程之间没有通信，这个是没有办法实现的

* 所以出现了两个唤醒机制：等待唤醒和通知唤醒

*/


//判断等待，业务，通知

public class SynchronizedResource {

private int num = 0;



//只要涉及到了并发编程就一定要加锁来保证安全性

//+1

public synchronized void increment() throws InterruptedException {

while (num != 0) {

//等待操作

this.wait();

}

num++;

//手动输出

System.out.println(Thread.currentThread().getName() + "=>" + num);

//通知其他线程这里执行完毕了

this.notifyAll();


}


//-1

public synchronized void decrement() throws InterruptedException {

while (num == 0) {

//等待操作

this.wait();

}

num--;

//手动输出

System.out.println(Thread.currentThread().getName() + "=>" + num);

//通知其他线程这里执行完毕了

this.notifyAll();


}


}


class Main1 {


public static void main(String[] args) {


SynchronizedResource synchronizedResource = new SynchronizedResource();


new Thread(()->{

for (int i = 0;i < 10;i++){

try {

synchronizedResource.increment();

} catch (InterruptedException e) {

e.printStackTrace();

}

}

},"A").start();



new Thread(()->{

for (int i = 0;i < 10;i++){

try {

synchronizedResource.decrement();

} catch (InterruptedException e) {

e.printStackTrace();

}

}


},"B").start();


new Thread(()->{

for (int i = 0;i < 10;i++){

try {

synchronizedResource.increment();

} catch (InterruptedException e) {

e.printStackTrace();

}

}


},"C").start();


new Thread(()->{

for (int i = 0;i < 10;i++){

try {

synchronizedResource.decrement();

} catch (InterruptedException e) {

e.printStackTrace();

}

}


},"D").start();

}


}
