//input事件解决限制长度的问题，onKeypress解决可以输入字母e的问题

<el-input

type="number"

v-model="form.number"

@input.native="changeNum"

onKeypress="return(/[\d]/.test(String.fromCharCode(event.keyCode)))"

>

</el-input>


changeNum() {

//6是限制的长度

if (this.form.number.length > 6) {

this.form.number = this.form.number.slice(0, 6);

}

},
