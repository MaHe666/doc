package com.mahe666.springcloud.controller;


import lombok.extern.slf4j.Slf4j;

import org.springframework.web.bind.annotation.GetMapping;

import org.springframework.web.bind.annotation.RequestMapping;

import org.springframework.web.bind.annotation.RestController;

import org.springframework.web.client.RestTemplate;


import javax.annotation.Resource;


@RestController

@Slf4j

@RequestMapping("/consumer")

public class ConsumerController {

//请求地址

private static final String INVOKE_URL = "http://cloud-zookeeper-provider-service/provider";


@Resource

private RestTemplate restTemplate;


@GetMapping("/get/zookeeperInfo")

public String getZookeeperInfo(){

String res = restTemplate.getForObject(INVOKE_URL + "/get/zookeeperInfo", String.class);

return res;

}



}
