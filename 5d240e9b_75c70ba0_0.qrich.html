/**

* 真正的多线程开发，公司中的开发，降低耦合性

*

* 线程就是一个单独的资源类，没有任何附属的操作

* 1，属性

* 2，方法

*/

public class SaleTicketDemo01 {

public static void main(String[] args) {

//并发，多线程操作同一个资源类

//需要把资源类丢进线程中

Ticket1 ticket = new Ticket1();


//@FunctionalInterface 函数式接口


new Thread(() -> {

for (int i = 0; i < 60; i++) {

ticket.sale();

}

},"A").start();


new Thread(() -> {

for (int i = 0; i < 60; i++) {

ticket.sale();

}

},"B").start();


new Thread(() -> {

for (int i = 0; i < 60; i++) {

ticket.sale();

}

},"C").start();


}

}
