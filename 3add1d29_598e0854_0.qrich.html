package com.mahe666.springcloud.controller;


import com.alibaba.csp.sentinel.annotation.SentinelResource;

import com.alibaba.csp.sentinel.slots.block.BlockException;

import org.springframework.web.bind.annotation.RequestMapping;

import org.springframework.web.bind.annotation.RestController;


@RestController

public class HelloWorld3Controller {


/**

* 注解方式调用资源

* @SentinelResource value 资源名称

* @SentinelResource blockHandler 调用被限流/降级/系统保护时调用的方法

*/

@RequestMapping("/helloworld3")

//value是流控规则名称，blockHandler是备选方法

@SentinelResource(value = "HelloWorld3",blockHandler = "blockHandlerForHelloWorld3")

public String helloworld3(){

return "Hello Sentinel,this is annotation way,@SentinelResource"+System.currentTimeMillis();

}



/**

* 原方法调用被限流/降级/系统保护时调用

* @param ex

* @return

*/

public String blockHandlerForHelloWorld3(BlockException ex){

ex.printStackTrace();

return "系统繁忙，请稍后";

}



}
