<template>

<h1>{{ msg }}</h1>

</template>


<script>

import axios from 'axios'


import Hello2 from "../../api/Hello2"


export default {

name: "Hello2",

data() {

return {

msg: 'Hello2'

}

},

methods: {

hello2_get() {

axios.get('http://localhost:8080/demo/hello2', {

params: {

"msg": this.msg

}

})

.then((resp) => {

console.log(resp)

this.msg = resp.data.name;

})

.catch((error) => {

console.log(error)

})

},

hello2_post() {

Hello2.printHelloWorld({

msg: this.msg

}).then((resp) => {

console.log(resp)

this.msg = resp.data.name;

}).catch((error) => {

console.log(error)

})

},


},

created() {

//this.hello2_get();

this.hello2_post();

}

}

</script>
