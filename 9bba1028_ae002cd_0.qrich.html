public class Example implements Comparable<Example> {


private int num;


public int getNum() {

return num;

}


public void setNum(int num) {

this.num = num;

}


@Override

public int compareTo(Example o) {

return this.getNum() - o.getNum();

}


@Override

public String toString() {

return "Example{" +

"num=" + num +

'}';

}

}
