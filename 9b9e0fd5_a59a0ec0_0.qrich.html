package com.mahe666.annotation;


import java.lang.annotation.*;


//元注解的作用就是负责注解其他注解（自定义的）

@myAnnotation

public class demo01 {


@myAnnotation

public void test(){


}

}


/*

* Target注解 表示我们的注解可以用在哪些地方

*

* 注解的属性值 可以让被他修饰的注解 修饰方法或类

* ElementType.METHOD 修饰方法

* ElementType.TYPE 修饰类

*/

@Target(value={ElementType.METHOD,ElementType.TYPE})


/*

* Retention注解 表示我们的注解在哪里生效

*

* Runtime代表运行时有效

* Class代表在编译后的文件中有效

* Source代表在源文件中有效

*

* Runtime > Class > Source

* 值为Runtime时，注解在Class和Source中也有效

* 值为Class时，注解在Source中也有效

*/

@Retention(value= RetentionPolicy.RUNTIME)


/*

* Documented注解 表示是否将我们的注解生成在javadoc中

*/

@Documented


/*

* Inherited注解 表示子类可以继承父类的注解

*/

@Inherited

//定义一个注解

@interface myAnnotation{


}
