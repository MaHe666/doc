package com.mahe666.single;


/**

* 饿汉式 单例模式

* 饿汉式的特点是 一上来就把这个对象给加载了

* 但是这里有一个问题：这个单例不用的时候有可能浪费内存

*

* 所以就出现了第二种说法 - 懒汉式

*/

public class HungryMan {


//构造方法私有，外部无法new对象

private HungryMan(){

}


// 创建全局的唯一一个对象

private final static HungryMan HUNGRY_MAN = new HungryMan();


// 对外抛出一个方法来获取对象

public static HungryMan getInstance(){

return HUNGRY_MAN;

}


}
