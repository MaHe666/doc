package com.mahe666.fourInterfaces;


import java.util.function.Function;


public class FunctionTest {

public static void main(String[] args) {

// 该接口下的apply方法只有一个作用：输出输入的值


//正常使用

Function function1 = new Function<String, String>() {

@Override

public String apply(String s) {

return s;

}

};


//使用lambda表达式

Function function2 = (str)->{return str;};

//Function function2 = str -> str;


System.out.println(function1.apply("HelloWorld"));

System.out.println(function2.apply("HelloWorld"));


}

}
