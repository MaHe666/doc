package com.mahe666.springcloud.config;


import org.springframework.context.annotation.Bean;

import org.springframework.context.annotation.Configuration;

import org.springframework.web.client.RestTemplate;


@Configuration

public class ConfigBean {


//RestTemplate提供了多种便捷访问远程Http服务的方法，是一种简单便捷的访问restful服务模板类，

//是Spring提供的用于访问Rest服务的客户端模板工具集

//使用RestTemplate访问restful接口非常的简粗暴且无脑

//（url，requsetMap，ResponseBean.class） 这三个参数分别代表REST请求地址，请求参数，Http响应转换 被 转换成的对象类型

@Bean

public RestTemplate getRestTemplate(){

return new RestTemplate();

}


}
