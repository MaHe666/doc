package com.mahe666.mapper;


import com.baomidou.mybatisplus.core.mapper.BaseMapper;

import com.mahe666.pojo.User;

import org.apache.ibatis.annotations.Mapper;

import org.springframework.stereotype.Repository;


/*

Mybatis-plus不需要xml映射文件

在对应的Mapper上面继承基本的接口 BaseMapper 就好

BaseMapper 有范型，我们传入的具体的类型会替代所有原接口内的泛型

*/


//可以在这里添加@Mapper注解表示它是Mapper层

//也可以在这里写@Repository注解表示它是dao层，代表它是持久层的

@Repository

public interface UserMapper extends BaseMapper<User> {

//所有的CRUD已经编写完成，不需要像以前那样的配置一大堆文件了

}
