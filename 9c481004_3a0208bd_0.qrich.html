public class Demo03 {

public static void main(String[] args) {

char c1 = 'a';

char c2 = '中';


System.out.println(c1);

System.out.println(c2);


//所有的字符本质上还是数字

System.out.println((int)c1);

System.out.println((int)c2);


//这里涉及到了编码的问题，Unicode编码可以处理任何语言的文字，他占了2个字节


char c3 = '\u0061';

System.out.println(c3);

}

}
