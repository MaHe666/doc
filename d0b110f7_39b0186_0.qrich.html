package com.mahe666.forkjoin;


import java.util.OptionalLong;

import java.util.concurrent.ExecutionException;

import java.util.concurrent.ForkJoinPool;

import java.util.concurrent.ForkJoinTask;

import java.util.stream.LongStream;


public class Main {

public static void main(String[] args) throws ExecutionException, InterruptedException {

test1();

test2();

test3();

test4();

}


public static void test1() {

long start = System.currentTimeMillis();

long sum = 0L;

for (long i = 1; i <= 10_0000_0000; i++) {

sum += i;

}

long end = System.currentTimeMillis();

System.out.println("sum="+sum+" 耗时="+(end - start));

}


public static void test2() throws ExecutionException, InterruptedException {

long start = System.currentTimeMillis();


ForkJoinPool pool = new ForkJoinPool();


//创建任务

ForkJoinTask<Long> task = new ForkjoinTest(1L,10_0000_0000L);


//同步提交任务，没有返回值

// new ForkJoinPool().execute(task);

//获取任务的返回值

// Long sum = task.get();


//异步提交任务，有返回值

ForkJoinTask<Long> submit = pool.submit(task);

//获取submit的返回值

Long sum = submit.get();


long end = System.currentTimeMillis();

System.out.println("sum="+sum+" 耗时="+(end - start));

}


// parallel()并行流

public static void test3() {

long start = System.currentTimeMillis();

// OptionalLong reduce = LongStream.rangeClosed(1L, 10_0000_0000L).parallel().reduce(Long::sum);

// Long sum = reduce.getAsLong();

long sum = LongStream.rangeClosed(1L, 10_0000_0000L).parallel().reduce(0,Long::sum);


long end = System.currentTimeMillis();

System.out.println("sum="+sum+" 耗时="+(end - start));

}


// sequential()串行流

public static void test4() {

long start = System.currentTimeMillis();

// OptionalLong reduce = LongStream.rangeClosed(1L, 10_0000_0000L).sequential().reduce(Long::sum);

// Long sum = reduce.getAsLong();

long sum = LongStream.rangeClosed(1L, 10_0000_0000L).sequential().reduce(0,Long::sum);


long end = System.currentTimeMillis();

System.out.println("sum="+sum+" 耗时="+(end - start));

}
