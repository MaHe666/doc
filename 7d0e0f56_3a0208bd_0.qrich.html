<div id="app">

<p> { { message | Reverse } } </p>

<p> { { message | Length } } </p>

</div>


<script src="https://cdn.jsdelivr.net/npm/vue/dist/vue.js"></script>

<script>

var demo = new Vue({

el: '#app',

data: {

message: '过滤器',

},

filters: {

Reverse: function(value) {

if (!value){

return '';

}

return value.toString().split('').reverse().join('');

},

Length: function(value) {

return value.length;

},

},

})

</script>
