package com.mahe666.annotation;


import java.lang.annotation.ElementType;

import java.lang.annotation.Retention;

import java.lang.annotation.RetentionPolicy;

import java.lang.annotation.Target;


public class demo02 {


@myAnnotation2(name = "mahe666",schools = {"清华大学","北京大学","皇家哈应用"})

public void test(){

}


@myAnnotation3("666")

public void test2(){


}


}


@Target({ElementType.METHOD,ElementType.TYPE})

@Retention(RetentionPolicy.RUNTIME)

@interface myAnnotation2{

/*

* 注解的参数

* 参数类型+参数名();

*

* 设置默认值

* 如果没有默认值而且在使用注解的时候不传参，注解会报错

* 参数类型+参数名() default 值

*

*/

String name() default "";

int age() default 0;

int id() default -1; //如果默认值为-1，则代表不存在

String[] schools() default {"哈工大","哈理工"};

}


@Target({ElementType.METHOD,ElementType.TYPE})

@Retention(RetentionPolicy.RUNTIME)

@interface myAnnotation3{

//如果参数值为value的情况下，可以省略value=

String value();

}
