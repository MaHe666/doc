<div id="app">

<mycomponent></mycomponent>

</div>


<!-- 组件模板 -->

<template id="mytemplate">

<div>

<h3>我是局部组件</h3>

<button @click="add">加</button>

</div>

</template>


<script src="https://cdn.jsdelivr.net/npm/vue/dist/vue.js"></script>

<script>

let obj = {

template: '#mytemplate', //使用选择器应用组件模板

data() {

return {

num: 1

}

},

methods: {

add() {

this.num++;

}

}

};


var vm = new Vue({

el: '#app',

components: { //声明局部组件

mycomponent: obj //组件名:组件实例

}

});

</script>
