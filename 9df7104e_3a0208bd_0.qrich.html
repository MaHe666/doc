package com.mahe666.controller;


import org.springframework.stereotype.Controller;

import org.springframework.ui.Model;

import org.springframework.web.bind.annotation.*;


@Controller


public class RestfulController {


@RequestMapping("/add1")

public String commonTest(int a, int b, Model model){

int res = a + b;

model.addAttribute("msg","结果为："+res);

return "test";

}



@RequestMapping("/add2/{a}/{b}")

//@PathVariable注解，让方法参数的值对应绑定到一个url模板变量上

public String restfulTest(@PathVariable int a,@PathVariable int b, Model model){

int res = a + b;

model.addAttribute("msg","结果为："+res);

return "test";

}


//可以使用@RequestMapping的method属性来约束请求method

//也可以使用@RequestMapping的变体来约束

/*

*/



//路径名相同时，会走对应提交方式的那个

//但是两个条件不要同时满足

@PostMapping("/add2/{a}/{b}")

public String postTest(@PathVariable int a,@PathVariable int b, Model model){

int res = a + b;

model.addAttribute("msg","结果为："+res);

return "test";

}


@GetMapping("/add2/{a}/{b}")

public String getTest(@PathVariable int a,@PathVariable int b, Model model){

int res = a + b;

model.addAttribute("msg","结果为："+res);

return "test";

}



}
