import pandas as pd


#当列表头上面有错误信息时

#读取excel表格，并封装为对象,header属性可以指定某一行为列表头，默认为0

df = pd.read_excel('./output.xlsx',header=0)

df.set_index('id',inplace=True)


#输出修改列表头的表格

df.to_excel('./output2.xlsx')
