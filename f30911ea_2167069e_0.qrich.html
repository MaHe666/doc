<template>

<button v-on:click="clickMethod">已经点击了 {{count}} 次按钮</button>

</template>


<script>

export default {

name: "buttonCounter",

props:{

value:{

type: Number,

}

},

data(){

return{

count: this.value

}

},

watch:{

count(val){

this.$emit("input", val)

}

},

methods:{

clickMethod(){

this.count++

this.$emit("input", this.count)

}

},

}

</script>
