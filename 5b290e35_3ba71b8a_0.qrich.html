package com.mahe666.rw;


import java.util.HashMap;

import java.util.Map;

import java.util.concurrent.locks.ReadWriteLock;

import java.util.concurrent.locks.ReentrantReadWriteLock;


/**

* 写锁是独占锁，一次只能被一个线程占有

* 共享锁共享锁，多个线程可以同时占有

*/

public class ReadWriteLockTest {

public static void main(String[] args) {

MyCacheWithRWLock myCache = new MyCacheWithRWLock();


//这里只做一些写入的操作

for (int i = 1; i <= 5; i++) {

int tmp = i;


new Thread(()->{

myCache.put(tmp+"",tmp);

},String.valueOf(i)).start();

}


//这里只做读取的操作

for (int i = 1; i <= 5; i++) {

int tmp = i;

new Thread(()->{

myCache.get(tmp+"");

},String.valueOf(i)).start();

}


}

}


class MyCache{

private volatile Map<String,Object> map = new HashMap<>();


//存放。写入

public void put(String key,Object value){

System.out.println(Thread.currentThread().getName()+"-- 写入了 -->"+"key:"+key+" value:"+value);

//向map中写入

map.put(key,value);

System.out.println("写入成功");

}


//获取。读取

public void get(String key){

System.out.println(Thread.currentThread().getName()+"-- 读取了 -->"+key);

//获取map中的值

Object o = map.get(key);

System.out.println("读取成功");


}

}


//使用读写锁自定义缓存

class MyCacheWithRWLock{

private volatile Map<String,Object> map = new HashMap<>();

//创建一个读写锁，为了更加细粒度的控制

private ReadWriteLock readWriteLock = new ReentrantReadWriteLock();


//存放。写入

//加入了ReadWriteLock后，我希望这里同时只有一个线程可以写入

public void put(String key,Object value){

//加入写锁 readWriteLock.writeLock()

//上锁

readWriteLock.writeLock().lock();


try {

System.out.println(Thread.currentThread().getName()+"-- 写入了 -->"+"key:"+key+" value:"+value);

//向map中写入

map.put(key,value);

System.out.println("写入成功");

} catch (Exception e) {

e.printStackTrace();

} finally {

//解锁

readWriteLock.writeLock().unlock();

}

}


//获取。读取

//加入了ReadWriteLock后，我希望这里所有线程都可以读

public void get(String key){

//加入读锁 readWriteLock.readLock()

//上锁

readWriteLock.readLock().lock();


try {

System.out.println(Thread.currentThread().getName()+"-- 读取了 -->"+key);

//获取map中的值

Object o = map.get(key);

System.out.println("读取成功");

} catch (Exception e) {

e.printStackTrace();

} finally {

//解锁

readWriteLock.readLock().unlock();

}



}

}
