package com.mahe666.single;


/**

* 懒汉式 单例模式

*/

public class LazyMan {


private LazyMan() {


}


//先不加载，第一次使用再加载

private static LazyMan lazyMan;


//返回示例

public static LazyMan getInstance(){

//进行验证判断

if (lazyMan == null){

lazyMan = new LazyMan();

}

return lazyMan;

}


// 单线程下上述代码没有问题，但是在多线程下就会出问题了

public static void main(String[] args) {

for (int i = 0; i < 10; i++) {

new Thread(()->{

// 直接去看对象的hashcode，发现实现单例的次数很少

// 所以我们需要修改我们 getInstance() 方法的代码

System.out.println(LazyMan.getInstance().hashCode());

}).start();

}

}

}
