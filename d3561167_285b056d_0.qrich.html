package com.mahe666;


import com.baomidou.mybatisplus.annotation.DbType;

import com.baomidou.mybatisplus.annotation.FieldFill;

import com.baomidou.mybatisplus.annotation.IdType;

import com.baomidou.mybatisplus.generator.AutoGenerator;

import com.baomidou.mybatisplus.generator.config.DataSourceConfig;

import com.baomidou.mybatisplus.generator.config.GlobalConfig;

import com.baomidou.mybatisplus.generator.config.PackageConfig;

import com.baomidou.mybatisplus.generator.config.StrategyConfig;

import com.baomidou.mybatisplus.generator.config.po.TableFill;

import com.baomidou.mybatisplus.generator.config.rules.DateType;

import com.baomidou.mybatisplus.generator.config.rules.NamingStrategy;


import java.util.ArrayList;


//代码自动生成器

public class MyGenerateCode {

public static void main(String[] args) {

//需要构建一个代码自动生成器 对象

AutoGenerator mpg = new AutoGenerator();

//在这里我们需要配置一些策略


////全局配置


//注意，这里需要是generator下的包

GlobalConfig gc = new GlobalConfig();


//获取项目路径

String projectPath = System.getProperty("user.dir");


//设置输出目录，生成到项目的指定路径下

gc.setOutputDir(projectPath + "/src/main/java");


//设置作者的名字，生成的代码里面都会有作者名字的注释

gc.setAuthor("作者名字");

//是否打开资源管理器

gc.setOpen(false);

//是否覆盖原来生成的文件

gc.setFileOverride(false);

//生成Service时，去掉I前缀，%s是正则表达式

gc.setServiceName("%sService");

//设置id生成策略

gc.setIdType(IdType.ID_WORKER);

//设置日期的类型，仅仅只是时间

gc.setDateType(DateType.ONLY_DATE);

//设置Swagger文档

gc.setSwagger2(true);


//加载配置

mpg.setGlobalConfig(gc);



////数据源配置

DataSourceConfig dsc = new DataSourceConfig();


//配置驱动

dsc.setDriverName("数据库驱动");

//配置路径

dsc.setUrl("数据库连接");

//配置用户名

dsc.setUsername("数据库用户名");

//配置密码

dsc.setPassword("数据库密码");

//设置数据库类型

dsc.setDbType(DbType.MYSQL);


//加载配置

mpg.setDataSource(dsc);



///包的配置

PackageConfig pc = new PackageConfig();


//生成一个模块

pc.setModuleName("模块名");

//生成对应的GroupId文件夹（哪个包下）

pc.setParent("org.example");

//设置实体类的名字（包名）

pc.setEntity("pojo");//或者Entity

//设置Mapper层的名字（包名）

pc.setMapper("mapper");

//设置Service层的名字（包名）

pc.setService("service");

//设置Controller层的名字（包名）

pc.setController("controller");


//加载配置

mpg.setPackageInfo(pc);



////策略配置

StrategyConfig sc = new StrategyConfig();


//设置一些需要映射的表名

sc.setInclude("表名1","表名2","表名3");


//设置包命名的一些规则，下划线转驼峰

sc.setNaming(NamingStrategy.underline_to_camel);

//设置列（数据库字段）的名字，下划线转驼峰

sc.setColumnNaming(NamingStrategy.underline_to_camel);


//自动生成Lombok

sc.setEntityLombokModel(true);


//逻辑删除，设置逻辑删除的字段名

sc.setLogicDeleteFieldName("关联的名字");


//自动填充配置

TableFill gmtCreate = new TableFill("gmt_create", FieldFill.INSERT);

TableFill gmtModify = new TableFill("gmt_modify", FieldFill.INSERT_UPDATE);

//创建需要添加到配置中的列表

ArrayList<TableFill> tableFills = new ArrayList<>();

tableFills.add(gmtCreate);

tableFills.add(gmtModify);

//加载自动填充配置到策略配置

sc.setTableFillList(tableFills);


//乐观锁的配置

sc.setVersionFieldName("version");


//开启restful的驼峰命名格式

sc.setRestControllerStyle(true);

//将请求转换为下划线

// localhost:8080/hello_id_2

sc.setControllerMappingHyphenStyle(true);


mpg.setStrategy(sc);



//执行代码构造器

mpg.execute();



}




}
