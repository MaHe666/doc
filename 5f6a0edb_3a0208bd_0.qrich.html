package com.mahe666.aba;


import java.util.concurrent.TimeUnit;

import java.util.concurrent.atomic.AtomicReference;

import java.util.concurrent.atomic.AtomicStampedReference;


public class Demo01 {

public static void main(String[] args) {

//普通的原子引用

// new AtomicReference<>();

//带时间的原子引用，这个时间戳就相当于版本号

// new AtomicStampedReference<>();


//初始值为0，初始时间戳为1

AtomicStampedReference<Integer> reference = new AtomicStampedReference<>(0,1);


new Thread(()->{

int stamp = reference.getStamp(); // 获得时间戳（版本号）

System.out.println("A1=>"+stamp);


// 为了顺序是我们想要的那样，这里是一个延迟

try {

TimeUnit.SECONDS.sleep(1);

} catch (InterruptedException e) {

e.printStackTrace();

}


// 修改值，四个参数分别为，期待值，更改值，期待时间戳，更改时间戳。两个时间戳位置就像期待值和更改值那样

System.out.println(reference.compareAndSet(0, 1, reference.getStamp(), reference.getStamp() + 1));

System.out.println("A2=>"+reference.getStamp());

// 再修改回来

System.out.println(reference.compareAndSet(1, 0, reference.getStamp(), reference.getStamp() + 1));

System.out.println("A3=>"+reference.getStamp());


},"A").start();


new Thread(()->{

int stamp = reference.getStamp(); // 获得时间戳（版本号）

System.out.println("B1=>"+stamp);


// 为了顺序是我们想要的那样，这里是一个延迟

try {

TimeUnit.SECONDS.sleep(2);

} catch (InterruptedException e) {

e.printStackTrace();

}


// 使用的是2秒前获取的stamp

System.out.println(reference.compareAndSet(0, 3, stamp, stamp + 1));


System.out.println("B2=>"+reference.getStamp());


},"B").start();


}

}
